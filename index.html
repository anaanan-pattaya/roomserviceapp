${groupName}

`; const grid = groupSection.querySelector('.grid-container'); groups[groupName].forEach(item => { // ดักสถานะตัวพิมพ์เล็ก-ใหญ่ if (item.status && item.status.trim().toLowerCase() !== "available") return; const card = document.createElement('div'); card.className = "bg-gray-800 border border-gray-700 border-opacity-60 rounded-2xl p-4 flex space-x-4 items-center shadow-sm"; const imgHtml = item.image_url ? `` : ''; let menuTitle = (currentLang === "TH") ? (item.menu_name_th || item.menu_name_en) : (item.menu_name_en || item.menu_name_th); let menuDesc = (currentLang === "TH") ? (item.description_th || "") : (item.description_en || ""); // เอา Sub group ออกเรียบร้อยแล้ว จะโชว์แค่ รูป -> ชื่ออาหาร -> รายละเอียด -> ราคา ครับน้า card.innerHTML = ` ${imgHtml}

${menuTitle}

${menuDesc}

฿${item.price_thb}
`; grid.appendChild(card); }); container.appendChild(groupSection); } }
